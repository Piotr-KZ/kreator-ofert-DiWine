# 1. ENCRYPTION — PEŁNY CYKL ŻYCIA CREDENTIALS
# =============================================================================

class TestEncryptionE2E:
    """
    Scenariusz: Pani Zosia dodaje integrację Gmail.
    AutoFlow szyfruje credentials przez fasthub_core, zapisuje, odczytuje, deszyfruje.
    """

    def test_full_credential_lifecycle(self, fasthub_encryption_key):
        """Pełny cykl: create → encrypt → store → retrieve → decrypt → verify."""
        from fasthub_core.security.encryption import (
            encrypt_credentials, decrypt_credentials, is_encryption_available
        )

        assert is_encryption_available() == True

        # Credentials integracji Gmail
        original_creds = {
            "email": "zosia@firma.pl",
            "app_password": "xyzw-abcd-efgh-ijkl",
            "imap_server": "imap.gmail.com",
            "smtp_server": "smtp.gmail.com",
        }

        # Szyfrowanie
        encrypted = encrypt_credentials(original_creds)
        assert encrypted.startswith("ENC:")
        assert "zosia@firma.pl" not in encrypted
        assert "xyzw-abcd-efgh-ijkl" not in encrypted

        # Symulacja zapisu do bazy (jako string)
        stored_value = encrypted
        assert isinstance(stored_value, str)

        # Odczyt i deszyfrowanie
        decrypted = decrypt_credentials(stored_value)
        assert decrypted == original_creds
        assert decrypted["email"] == "zosia@firma.pl"
        assert decrypted["app_password"] == "xyzw-abcd-efgh-ijkl"

    def test_encryption_key_rotation(self, fasthub_encryption_key):
        """Admin rotuje klucz szyfrowania — stare dane muszą być przeniesione."""
        from fasthub_core.security.encryption import (
            encrypt_credentials, decrypt_credentials,
            rotate_encryption_key, generate_key
        )

        old_key = fasthub_encryption_key
        new_key = generate_key()

        # Szyfruj starym kluczem
        creds = {"api_key": "sk-production-key-12345"}
        encrypted_old = encrypt_credentials(creds)

        # Rotacja
        encrypted_new = rotate_encryption_key(old_key, new_key, encrypted_old)

        # Nowy klucz deszyfruje
        os.environ["FASTHUB_SECRET_KEY"] = new_key
        decrypted = decrypt_credentials(encrypted_new)
        assert decrypted["api_key"] == "sk-production-key-12345"

        # Przywróć stary klucz dla reszty testów
        os.environ["FASTHUB_SECRET_KEY"] = old_key

    def test_mask_credentials_for_logs(self):
        """Credentials w logach muszą być zamaskowane."""
        from fasthub_core.security.encryption import mask_credentials

        creds = {
            "api_key": "sk-very-long-secret-key-12345",
            "webhook_secret": "whsec_abcdefgh",
            "name": "Moja integracja Fakturownia",
            "url": "https://app.fakturownia.pl",
        }

        masked = mask_credentials(creds)

        # Sekrety zamaskowane
        assert "sk-very-long-secret-key-12345" not in str(masked)
        assert "whsec_abcdefgh" not in str(masked)

        # Publiczne dane zachowane
        assert masked["name"] == "Moja integracja Fakturownia"
        assert masked["url"] == "https://app.fakturownia.pl"

    def test_encryption_via_thin_wrapper(self, fasthub_encryption_key):
        """Import przez stary path (thin wrapper) daje identyczny wynik."""
        from app.core.encryption import encrypt_credentials as old_encrypt
        from fasthub_core.security.encryption import encrypt_credentials as new_encrypt

        # Muszą być tą samą funkcją (re-export)
        assert old_encrypt is new_encrypt

        # I działać tak samo
        creds = {"token": "test-123"}
        enc1 = old_encrypt(creds)
        assert enc1.startswith("ENC:")

# 2. EVENT BUS — EMISJA I PROPAGACJA EVENTÓW

class TestEventBusE2E:
    """
    Scenariusz: Proces AutoFlow kończy się sukcesem.
    Event Bus emituje process.completed → handlery to łapią.
    """

    def test_event_emission_and_handling(self, fresh_event_bus):
        """Emit → handler → payload dostępny."""
        received_events = []

        async def capture_handler(event):
            received_events.append(event)

        fresh_event_bus.register("process.*", capture_handler)

        run_async(fresh_event_bus.emit("process.completed", {
            "process_id": 42,
            "execution_id": "exec-abc-123",
            "duration_ms": 3450,
            "steps_completed": 5,
        }))

        assert len(received_events) == 1
        assert received_events[0]["payload"]["process_id"] == 42
        assert received_events[0]["payload"]["duration_ms"] == 3450

    def test_wildcard_catches_multiple_events(self, fresh_event_bus):
        """Wildcard handler łapie wszystkie eventy z kategorii."""
        caught = []

        fresh_event_bus.register("execution.*", lambda e: caught.append(e))

        run_async(fresh_event_bus.emit("execution.started", {"id": 1}))
        run_async(fresh_event_bus.emit("execution.failed", {"id": 2, "error": "timeout"}))
        run_async(fresh_event_bus.emit("execution.completed", {"id": 3}))

        # Powinien złapać 3 eventy
        assert len(caught) == 3

    def test_event_bus_does_not_cross_categories(self, fresh_event_bus):
        """Handler na execution.* NIE łapie process.* eventów."""
        caught = []

        fresh_event_bus.register("execution.*", lambda e: caught.append(e))

        run_async(fresh_event_bus.emit("process.completed", {"id": 1}))
        run_async(fresh_event_bus.emit("billing.limit_exceeded", {"id": 2}))

        assert len(caught) == 0

    def test_billing_limit_event(self, fresh_event_bus):
        """Event billing.limit_approaching emitowany gdy usage > 80%."""
        alerts = []

        fresh_event_bus.register("billing.*", lambda e: alerts.append(e))

        # Symulacja: billing middleware emituje event
        run_async(fresh_event_bus.emit("billing.limit_approaching", {
            "tenant_id": "firma-abc",
            "resource": "executions",
            "current": 420,
            "limit": 500,
            "percentage": 84,
        }))

        assert len(alerts) == 1
        assert alerts[0]["payload"]["percentage"] == 84

    def test_multiple_handlers_same_event(self, fresh_event_bus):
        """Wiele handlerów na ten sam event — wszystkie wywołane."""
        log1, log2, log3 = [], [], []

        fresh_event_bus.register("notification.created", lambda e: log1.append(1))
        fresh_event_bus.register("notification.created", lambda e: log2.append(2))
        fresh_event_bus.register("notification.*", lambda e: log3.append(3))

        count = run_async(fresh_event_bus.emit("notification.created", {"msg": "test"}))

        assert len(log1) == 1
        assert len(log2) == 1
        assert len(log3) == 1
        assert count >= 3

    def test_event_structure(self, fresh_event_bus):
        """Event ma wymagane pola: event_type, payload, timestamp, source."""
        captured = []
        fresh_event_bus.register("test.*", lambda e: captured.append(e))

        run_async(fresh_event_bus.emit("test.structure", {"key": "val"}, source="api"))

        event = captured[0]
        assert "event_type" in event
        assert "payload" in event
        assert "timestamp" in event
        assert "source" in event
        assert event["source"] == "api"

    def test_autoflow_specific_handlers_registration(self):
        """register_autoflow_handlers() z thin wrappera działa."""
        from fasthub_core.events.bus import EventBus
        bus = EventBus()
        bus.clear_handlers()

        # Zaimportuj i zarejestruj
        from app.events import register_autoflow_handlers
        register_autoflow_handlers()

        # Bus powinien mieć handlery
        stats = bus.get_stats()
        # Uwaga: register_autoflow_handlers rejestruje na globalnym event_bus,
        # nie na naszym lokalnym bus — sprawdź z globalnego
        from fasthub_core.events.bus import event_bus
        global_stats = event_bus.get_stats()
        assert global_stats["handlers_count"] >= 1

class TestOAuthE2E:
    """
    Scenariusz: Pani Zosia podłącza Google Drive do AutoFlow.

    """

    def test_authorization_url_generation(self, sample_oauth_config):
        """Generacja URL autoryzacji z PKCE."""
        from fasthub_core.integrations.oauth import OAuthManager, MemoryTokenStorage

        manager = OAuthManager(

        assert "accounts.google.com" in url

        assert "scope=" in url
        assert len(state) > 10  # random state string

        # PKCE enabled — URL powinien mieć code_challenge
        if sample_oauth_config.pkce_enabled:
            assert "code_challenge=" in url
            assert "code_challenge_method=S256" in url

    def test_token_storage_memory(self, sample_oauth_config):
        """Tokeny zapisywane i odczytywane z MemoryTokenStorage."""
        from fasthub_core.integrations.oauth import (
            OAuthTokens, MemoryTokenStorage
        )

        storage = MemoryTokenStorage()
        tokens = OAuthTokens(
            access_token="ya29.test-access-token",
            refresh_token="1//test-refresh-token",
            expires_at=datetime.utcnow() + timedelta(hours=1),
            token_type="Bearer",
            scope="email profile",
        )

        # Save
        run_async(storage.save_tokens("google", "integration-42", tokens))

        # Retrieve
        loaded = run_async(storage.get_tokens("google", "integration-42"))
        assert loaded is not None
        assert loaded.access_token == "ya29.test-access-token"
        assert loaded.refresh_token == "1//test-refresh-token"
        assert loaded.scope == "email profile"
        assert loaded.is_expired() == False

    def test_token_expiry_detection(self):
        """Tokeny expired → should_refresh() = True."""
        from fasthub_core.integrations.oauth import OAuthTokens

        expired_token = OAuthTokens(
            access_token="expired-token",
            refresh_token="refresh",
            expires_at=datetime.utcnow() - timedelta(minutes=5),
        )
        assert expired_token.is_expired() == True
        assert expired_token.should_refresh() == True

        # Token expiring soon (within buffer)
        soon_token = OAuthTokens(
            access_token="soon-token",
            refresh_token="refresh",
            expires_at=datetime.utcnow() + timedelta(minutes=3),
        )
        assert soon_token.is_expired() == False
        assert soon_token.should_refresh(buffer_seconds=300) == True  # 5min buffer

    def test_token_serialization_roundtrip(self):
        """Tokeny → dict → tokeny (do zapisu w DB jako JSON)."""
        from fasthub_core.integrations.oauth import OAuthTokens

        original = OAuthTokens(
            access_token="acc_test",
            refresh_token="ref_test",
            expires_at=datetime.utcnow() + timedelta(hours=1),
            token_type="Bearer",
            scope="read write",
        )

        # To dict (symulacja zapisu)
        d = original.to_dict()
        assert isinstance(d, dict)
        assert d["access_token"] == "acc_test"

        # From dict (symulacja odczytu)
        restored = OAuthTokens.from_dict(d)
        assert restored.access_token == "acc_test"
        assert restored.refresh_token == "ref_test"
        assert restored.scope == "read write"

# 4. WEBHOOKS — WERYFIKACJA PODPISÓW
# =============================================================================

class TestWebhooksE2E:
    """
    Scenariusz: Stripe wysyła webhook do AutoFlow.
    AutoFlow weryfikuje podpis HMAC-SHA256 przed przetworzeniem.
    """

    def test_stripe_webhook_signature_flow(self):
        """Pełny flow: compute → send → verify."""
        from fasthub_core.integrations.webhooks import SignatureVerifier, SignatureMethod

        webhook_secret = "whsec_test_stripe_secret_2024"
        payload = json.dumps({
            "id": "evt_1234567890",
            "type": "invoice.payment_succeeded",
            "data": {
                "object": {
                    "customer": "cus_abc",
                    "amount_paid": 9900,
                    "currency": "pln",
                }
            }
        }).encode()

        # Stripe oblicza podpis
        signature = SignatureVerifier.compute_signature(
            payload, webhook_secret, SignatureMethod.HMAC_SHA256
        )

        # AutoFlow weryfikuje
        is_valid = SignatureVerifier.verify_signature(
            payload, webhook_secret, signature, SignatureMethod.HMAC_SHA256
        )
        assert is_valid == True

    def test_tampered_payload_rejected(self):
        """Zmodyfikowany payload → podpis nieważny."""
        from fasthub_core.integrations.webhooks import SignatureVerifier, SignatureMethod

        secret = "my_webhook_secret"
        original_payload = b'{"amount": 100}'
        tampered_payload = b'{"amount": 99999}'

        sig = SignatureVerifier.compute_signature(
            original_payload, secret, SignatureMethod.HMAC_SHA256
        )

        # Oryginalny — OK
        assert SignatureVerifier.verify_signature(
            original_payload, secret, sig, SignatureMethod.HMAC_SHA256
        ) == True

        # Zmodyfikowany — FAIL
        assert SignatureVerifier.verify_signature(
            tampered_payload, secret, sig, SignatureMethod.HMAC_SHA256
        ) == False

    def test_wrong_secret_rejected(self):
        """Inny secret → podpis nieważny."""
        from fasthub_core.integrations.webhooks import SignatureVerifier, SignatureMethod

        payload = b'{"event": "test"}'
        sig = SignatureVerifier.compute_signature(
            payload, "correct_secret", SignatureMethod.HMAC_SHA256
        )

        assert SignatureVerifier.verify_signature(
            payload, "wrong_secret", sig, SignatureMethod.HMAC_SHA256
        ) == False

    def test_webhook_registration_and_dedup(self):
        """Rejestracja webhooka + dedup eventów."""
        from fasthub_core.integrations.webhooks import (
            MemoryWebhookStorage, WebhookRegistration, WebhookConfig,
            WebhookEvent, WebhookStatus, SignatureMethod,
        )

        storage = MemoryWebhookStorage()

        # Rejestracja
        config = WebhookConfig(
            url="https://firma-abc.autoflow.pl/webhooks/stripe",
            events=["invoice.paid", "subscription.updated"],
            secret="whsec_test",
            signature_method=SignatureMethod.HMAC_SHA256,
        )
        reg = WebhookRegistration(
            id="wh-stripe-1",
            config=config,
            status=WebhookStatus.ACTIVE,
        )
        run_async(storage.save_registration(reg))

        # Odczyt
        loaded = run_async(storage.get_registration("wh-stripe-1"))
        assert loaded is not None
        assert loaded.config.url == "https://firma-abc.autoflow.pl/webhooks/stripe"
        assert "invoice.paid" in loaded.config.events

        # Event + dedup
        event = WebhookEvent(
            id="evt-stripe-001",
            webhook_id="wh-stripe-1",
            event_type="invoice.paid",
            payload={"invoice_id": "inv_123", "amount": 9900},
            status="pending",
            attempts=0,
        )
        run_async(storage.save_event(event))

        # Pierwsza próba — nie duplikat
        is_dup = run_async(storage.check_duplicate("evt-stripe-001"))
        assert is_dup == True  # Już istnieje = duplikat

        # Nowy event — nie duplikat
        is_new = run_async(storage.check_duplicate("evt-stripe-999"))
        assert is_new == False

    def test_multiple_signature_methods(self):
        """HMAC-SHA256 i HMAC-SHA1 dają różne podpisy."""
        from fasthub_core.integrations.webhooks import SignatureVerifier, SignatureMethod

        payload = b"test payload"
        secret = "shared_secret"

        sha256_sig = SignatureVerifier.compute_signature(
            payload, secret, SignatureMethod.HMAC_SHA256
        )
        sha1_sig = SignatureVerifier.compute_signature(
            payload, secret, SignatureMethod.HMAC_SHA1
        )

        assert sha256_sig != sha1_sig

        # Każdy weryfikowalny swoją metodą
        assert SignatureVerifier.verify_signature(
            payload, secret, sha256_sig, SignatureMethod.HMAC_SHA256
        ) == True
        assert SignatureVerifier.verify_signature(
            payload, secret, sha1_sig, SignatureMethod.HMAC_SHA1
        ) == True

        # Ale nie zamiennie
        assert SignatureVerifier.verify_signature(
            payload, secret, sha256_sig, SignatureMethod.HMAC_SHA1
        ) == False

class TestCrossModuleE2E:
    """
    Scenariusze które przechodzą przez wiele modułów fasthub_core naraz.
    Najbliższe realnym use-case'om produkcyjnym.
    """

    def test_encrypted_oauth_tokens_in_storage(self, fasthub_encryption_key):
        """OAuth tokeny szyfrowane przed zapisem do storage."""
        from fasthub_core.integrations.oauth import OAuthTokens, MemoryTokenStorage
        from fasthub_core.security.encryption import encrypt_credentials, decrypt_credentials

        tokens = OAuthTokens(
            access_token="ya29.sensitive-access-token",
            refresh_token="1//sensitive-refresh",
            expires_at=datetime.utcnow() + timedelta(hours=1),
        )

        # Szyfruj przed zapisem
        token_dict = tokens.to_dict()
        encrypted = encrypt_credentials(token_dict)

        assert "ya29.sensitive-access-token" not in encrypted

        # Odszyfruj po odczycie
        decrypted_dict = decrypt_credentials(encrypted)
        restored = OAuthTokens.from_dict(decrypted_dict)

        assert restored.access_token == "ya29.sensitive-access-token"
        assert restored.is_expired() == False

    def test_event_bus_billing_encryption_flow(self, fresh_event_bus, fasthub_encryption_key):
        """
        Pełny scenariusz:
        1. Credentials zaszyfrowane
        2. Event billing.limit_approaching emitowany
        3. Handler łapie event
        """
        from fasthub_core.security.encryption import encrypt_credentials, is_encryption_available

        # 1. Szyfrowanie działa
        assert is_encryption_available() == True
        encrypted = encrypt_credentials({"stripe_key": "sk_live_xxx"})
        assert encrypted.startswith("ENC:")

        # 2. Event Bus emituje billing event
        billing_alerts = []
        fresh_event_bus.register("billing.*", lambda e: billing_alerts.append(e))

        run_async(fresh_event_bus.emit("billing.limit_approaching", {
            "tenant_id": "firma-xyz",
            "resource": "ai_operations",
            "current": 180,
            "limit": 200,
        }))

        # 3. Handler złapał
        assert len(billing_alerts) == 1
        assert billing_alerts[0]["payload"]["resource"] == "ai_operations"

    def test_webhook_verification_with_event_emission(self, fresh_event_bus):
        """
        Stripe webhook → verify signature → emit event.
        Symulacja pełnego flow obsługi Stripe webhook.
        """
        from fasthub_core.integrations.webhooks import SignatureVerifier, SignatureMethod

        processed_events = []
        fresh_event_bus.register("billing.*", lambda e: processed_events.append(e))

        # Symulacja Stripe webhook
        secret = "whsec_test"
        payload = json.dumps({"type": "invoice.paid", "amount": 9900}).encode()
        signature = SignatureVerifier.compute_signature(
            payload, secret, SignatureMethod.HMAC_SHA256
        )

        # Weryfikacja
        is_valid = SignatureVerifier.verify_signature(
            payload, secret, signature, SignatureMethod.HMAC_SHA256
        )
        assert is_valid == True

        # Po weryfikacji → emit event
        if is_valid:
            run_async(fresh_event_bus.emit("billing.invoice_paid", {
                "amount": 9900,
                "currency": "pln",
            }))

        assert len(processed_events) == 1
        assert processed_events[0]["payload"]["amount"] == 9900

### Jeśli testy failują

**ImportError z fasthub_core:**
- `pip install -e ../Fasthub` — upewnij się że jest zainstalowany
- Sprawdź czy Brief 10 wyeksportował daną funkcję

**AttributeError (brak metody/pola):**
- Nazwa mogła się zmienić w Brief 10 — sprawdź faktyczny kod fasthub_core i dopasuj test

**Encryption testy failują:**
- Sprawdź czy fixture `fasthub_encryption_key` ustawia `FASTHUB_SECRET_KEY`
- Sprawdź czy fasthub_core czyta ten env var

**Event Bus testy failują:**
- `fresh_event_bus` fixture musi tworzyć NOWY EventBus, nie globalny singleton
- Jeśli EventBus to singleton — użyj `clear_handlers()` przed każdym testem

### Dopasuj testy do rzeczywistości Brief 10

**KRYTYCZNE:** Brief 10 mógł zmienić nazwy funkcji, parametry klas, lub strukturę exportów. Testy w tym briefie są oparte na PLANOWANYM API z Brief 10. Jeśli Brief 10 zrobił coś inaczej — **DOPASUJ TESTY, NIE ZMIENIAJ FASTHUB_CORE.**

Przykłady możliwych rozbieżności:
- `BillingService.check_limit()` → mogło zostać `check_billing_limit()`
- `EventBus()` constructor → mogło wymagać parametrów
- `WebhookConfig` → mogło mieć inne wymagane pola
- `OAuthManager` → mogło mieć inną sygnaturę `__init__`

W każdym przypadku: przeczytaj faktyczny kod fasthub_core i dopasuj test.
