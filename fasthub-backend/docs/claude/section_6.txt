def test_replace_template_variables():
    """Test template variable replacement"""
    from app.workers.tasks.process_tasks import replace_template_variables
    
    config = {
        "subject": "Invoice {{step_1.number}}",
        "amount": "{{step_2.total}}",
        "nested": {
            "value": "{{step_1.name}}"
        }
    }
    
    context = {
        "step_1": {"number": "INV-001", "name": "Test"},
        "step_2": {"total": "1500"}
    }
    
    result = replace_template_variables(config, context)
    
    assert result["subject"] == "Invoice INV-001"
    assert result["amount"] == "1500"
    assert result["nested"]["value"] == "Test"


def test_get_from_context():
    """Test getting value from context using dot notation"""
    from app.workers.tasks.process_tasks import get_from_context
    
    context = {
        "step_1": {
            "email": {
                "subject": "Hello World"
            }
        }
    }
    
    value = get_from_context("step_1.email.subject", context)
    assert value == "Hello World"
    
    # Non-existent path
    value = get_from_context("step_1.email.body", context)
    assert value is None

@pytest.mark.integration
def test_celery_task_execution_integration():
    """
    Integration test - requires running Redis.
    
    Run with:
        pytest tests/test_celery_tasks.py::test_celery_task_execution_integration -v
    """
    from app.workers.tasks.email_tasks import send_verification_email
    
    # Execute task asynchronously
    task = send_verification_email.delay(
        email="integration@example.com",
        name="Integration Test",
        verification_url="https://test.com/verify"
    )
    
    # Wait for result (max 10 seconds)
    result = task.get(timeout=10)
    
    assert result["status"] == "sent"
    assert task.status == "SUCCESS"
