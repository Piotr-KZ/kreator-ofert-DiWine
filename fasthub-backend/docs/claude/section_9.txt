Cache Layers:

4. Application cache (computed results)
"""

import json
import pickle
from typing import Any, Optional, Callable
from functools import wraps
import redis
from app.core.config import settings
import hashlib

# Redis client
redis_client = redis.from_url(
    settings.REDIS_URL,
    decode_responses=False,  # Binary for pickle
    socket_connect_timeout=5,
    socket_keepalive=True,
    health_check_interval=30
)

# Redis client for JSON (strings)
redis_json_client = redis.from_url(
    settings.REDIS_URL,
    decode_responses=True,  # Strings for JSON
    socket_connect_timeout=5
)


class CacheNamespace:
    """Cache key namespaces for organization"""
    
    USER = "user"
    ORG = "org"
    SUBSCRIPTION = "sub"
    PROCESS = "proc"
    EXECUTION = "exec"
    INTEGRATION = "int"
    QUERY = "query"
    SESSION = "session"
    RATE_LIMIT = "ratelimit"


def make_cache_key(*parts: str) -> str:
    """
    Generate cache key from parts.
    
    Args:
        *parts: Key components
    
    Returns:
        Cache key string
    
    Example:
        make_cache_key("user", "123", "profile") -> "user:123:profile"
    """
    return ":".join(str(p) for p in parts)


def cache_get(key: str) -> Optional[Any]:
    """
    Get value from cache.
    
    Args:
        key: Cache key
    
    Returns:
        Cached value or None
    """
    try:
        value = redis_client.get(key)
        if value:
            return pickle.loads(value)
        return None
    except Exception as e:
        # Log error but don't fail
        print(f"Cache get error: {e}")
        return None


def cache_set(key: str, value: Any, ttl: int = 3600) -> bool:
    """
    Set value in cache.
    
    Args:
        key: Cache key
        value: Value to cache
        ttl: Time to live in seconds (default 1 hour)
    
    Returns:
        Success status
    """
    try:
        pickled = pickle.dumps(value)
        redis_client.setex(key, ttl, pickled)
        return True
    except Exception as e:
        print(f"Cache set error: {e}")
        return False


def cache_delete(key: str) -> bool:
    """Delete key from cache"""
    try:
        redis_client.delete(key)
        return True
    except Exception as e:
        print(f"Cache delete error: {e}")
        return False


def cache_delete_pattern(pattern: str) -> int:
    """
    Delete all keys matching pattern.
    
    Args:
        pattern: Key pattern (e.g., "user:123:*")
    
    Returns:
        Number of deleted keys
    """
    try:
        keys = redis_client.keys(pattern)
        if keys:
            return redis_client.delete(*keys)
        return 0
    except Exception as e:
        print(f"Cache delete pattern error: {e}")
        return 0


def cached(
    namespace: str,
    key_func: Optional[Callable] = None,
    ttl: int = 3600,
    cache_none: bool = False
):
    """
    Decorator for caching function results.
    
    Args:
        namespace: Cache namespace
        key_func: Function to generate cache key from args
        ttl: Time to live in seconds
        cache_none: Whether to cache None results
    
    Example:
        @cached(namespace="user", ttl=300)
        def get_user_profile(user_id: int):
            # Expensive database query
            return db.query(User).filter(User.id == user_id).first()
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Generate cache key
            if key_func:
                cache_key = key_func(*args, **kwargs)
            else:
                # Default: hash function name + args
                key_parts = [func.__name__]
                key_parts.extend(str(arg) for arg in args)
                key_parts.extend(f"{k}={v}" for k, v in sorted(kwargs.items()))
                key_str = ":".join(key_parts)
                key_hash = hashlib.md5(key_str.encode()).hexdigest()[:16]
                cache_key = make_cache_key(namespace, key_hash)
            
            # Try to get from cache
            cached_value = cache_get(cache_key)
            if cached_value is not None:
                return cached_value
            
            # Execute function
            result = func(*args, **kwargs)
            
            # Cache result
            if result is not None or cache_none:
                cache_set(cache_key, result, ttl)
            
            return result
        
        return wrapper
    return decorator


def invalidate_cache(namespace: str, *key_parts: str):
    """
    Invalidate cache for specific namespace/key.
    
    Example:
        invalidate_cache("user", "123", "*")  # Delete all user:123:* keys
    """
    pattern = make_cache_key(namespace, *key_parts)
    return cache_delete_pattern(pattern)

def check_rate_limit(
    identifier: str,
    max_requests: int,
    window_seconds: int
) -> tuple[bool, int]:
    """
    Check if request is within rate limit.
    
    Args:
        identifier: Unique identifier (user_id, IP, etc)
        max_requests: Max requests allowed
        window_seconds: Time window in seconds
    
    Returns:
        (allowed: bool, remaining: int)
    
    Example:
        allowed, remaining = check_rate_limit("user:123", 100, 60)
        if not allowed:
            raise HTTPException(429, "Too many requests")
    """
    key = make_cache_key(CacheNamespace.RATE_LIMIT, identifier)
    
    try:
        # Increment counter
        current = redis_json_client.incr(key)
        
        # Set expiry on first request
        if current == 1:
            redis_json_client.expire(key, window_seconds)
        
        # Check limit
        allowed = current <= max_requests
        remaining = max(0, max_requests - current)
        
        return allowed, remaining
    
    except Exception as e:
        print(f"Rate limit error: {e}")
        # On error, allow request (fail open)
        return True, max_requests

def warm_cache_for_organization(org_id: int):
    """
    Pre-warm cache for organization.
    
    Called on login or scheduled task.
    """
    from app.db.session import SessionLocal
    from app.db.models.organization import Organization
    from app.db.models.user import User
    from app.db.models.subscription import Subscription
    
    db = SessionLocal()
    
    try:
        # Cache organization
        org = db.query(Organization).filter(Organization.id == org_id).first()
        if org:
            cache_organization(org_id, {
                "id": org.id,
                "name": org.name,
                "slug": org.slug
            })
        
        # Cache active users
        users = db.query(User).filter(
            User.organization_id == org_id,
            User.is_active == True
        ).all()
        
        for user in users:
            cache_user(user.id, {
                "id": user.id,
                "email": user.email,
                "role": user.role
            })
        
        # Cache subscription
        subscription = db.query(Subscription).filter(
            Subscription.organization_id == org_id
        ).first()
        
        if subscription:
            key = make_cache_key(CacheNamespace.SUBSCRIPTION, str(org_id))
            cache_set(key, {
                "status": subscription.status,
                "plan": subscription.plan,
                "current_period_end": subscription.current_period_end.isoformat()
            }, ttl=1800)
    
    finally:
        db.close()

def get_cache_stats() -> dict:
    """
    Get cache statistics.
    
    Returns:
        {
            "keys": 1234,
            "memory_used": "15MB",
            "hit_rate": 0.85
        }
    """
    try:
        info = redis_client.info("stats")
        memory_info = redis_client.info("memory")
        
        hits = int(info.get("keyspace_hits", 0))
        misses = int(info.get("keyspace_misses", 0))
        total = hits + misses
        hit_rate = hits / total if total > 0 else 0
        
        return {
            "keys": redis_client.dbsize(),
            "memory_used_bytes": memory_info.get("used_memory", 0),
            "memory_used_human": memory_info.get("used_memory_human", "0B"),
            "hit_rate": round(hit_rate, 3),
            "total_requests": total,
            "hits": hits,
            "misses": misses
        }
    
    except Exception as e:
        print(f"Cache stats error: {e}")
        return {}

def get_db():
    """
    Database session dependency for FastAPI.
    
    Usage:
        @router.get("/users")
        def get_users(db: Session = Depends(get_db)):
            return db.query(User).all()
    """
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

"""
Database query optimization patterns.

Common problems:
1. N+1 queries
2. Missing indexes
3. Unoptimized joins
4. Large result sets
"""

from sqlalchemy.orm import joinedload, selectinload, contains_eager
from sqlalchemy import func, and_, or_
from typing import List, Optional
from app.db.models.user import User

# ============================================
# PROBLEM: N+1 QUERIES
# ============================================

def get_users_with_organizations_BAD(db):
    """
    ❌ BAD - N+1 queries
    
    Executes:
    1. SELECT * FROM users (1 query)
    2. SELECT * FROM organizations WHERE id = ? (N queries)
    
    Total: N+1 queries
    """
    users = db.query(User).all()
    
    for user in users:
        print(user.organization.name)  # Triggers separate query!
    
    return users


def get_users_with_organizations_GOOD(db):
    """
    ✅ GOOD - Single query with JOIN
    
    Executes:
    1. SELECT users.*, organizations.* FROM users JOIN organizations (1 query)
    
    Total: 1 query
    """
    users = db.query(User).options(
        joinedload(User.organization)  # Eager load organization
    ).all()
    
    for user in users:
        print(user.organization.name)  # No additional query!
    
    return users


# ============================================
# PROBLEM: MULTIPLE RELATIONSHIPS
# ============================================

def get_organizations_with_details_BAD(db):
    """
    ❌ BAD - Multiple N+1 queries
    
    For each organization:
    - Query users
    - Query subscription
    - Query processes
    
    Total: 1 + N + N + N queries
    """
    orgs = db.query(Organization).all()
    
    for org in orgs:
        print(len(org.users))  # N queries
        print(org.subscription.status)  # N queries
        print(len(org.processes))  # N queries
    
    return orgs


def get_organizations_with_details_GOOD(db):
    """
    ✅ GOOD - Eager load all relationships
    
    Uses selectinload for collections (better than joinedload for collections).
    
    Total: 4 queries (1 main + 3 for relationships)
    """
    orgs = db.query(Organization).options(
        selectinload(Organization.users),
        joinedload(Organization.subscription),
        selectinload(Organization.processes)
    ).all()
    
    for org in orgs:
        print(len(org.users))  # No query
        print(org.subscription.status)  # No query
        print(len(org.processes))  # No query
    
    return orgs

def get_all_executions_BAD(db):
    """
    ❌ BAD - Loads all records into memory
    
    Problems:
    - High memory usage
    - Slow response
    - Database load
    """
    executions = db.query(ProcessExecution).all()  # Could be millions!
    return executions


def get_all_executions_GOOD(db, page: int = 1, per_page: int = 50):
    """
    ✅ GOOD - Paginated results
    
    Benefits:
    - Low memory usage
    - Fast response
    - Reduced database load
    """
    offset = (page - 1) * per_page
    
    executions = db.query(ProcessExecution)\
        .order_by(ProcessExecution.started_at.desc())\
        .limit(per_page)\
        .offset(offset)\
        .all()
    
    # Get total count (for pagination metadata)
    total = db.query(func.count(ProcessExecution.id)).scalar()
    
    return {
        "items": executions,
        "page": page,
        "per_page": per_page,
        "total": total,
        "pages": (total + per_page - 1) // per_page
    }

class CacheControlMiddleware(BaseHTTPMiddleware):
    """
    Add Cache-Control headers for static/cacheable responses.
    """
    
    async def dispatch(self, request: Request, call_next: Callable):
        response = await call_next(request)
        
        # Cache static endpoints
        if request.url.path.startswith("/api/v1/public/"):
            response.headers["Cache-Control"] = "public, max-age=3600"
        
        # Don't cache authenticated endpoints
        elif "Authorization" in request.headers:
            response.headers["Cache-Control"] = "private, no-cache, no-store, must-revalidate"
        
        # Cache GET requests for 5 minutes
        elif request.method == "GET":
            response.headers["Cache-Control"] = "public, max-age=300"
        
        return response

from pydantic import BaseModel

class PaginatedResponse(BaseModel, Generic[T]):
    """
    Standard paginated response.
    
    Example:
        return PaginatedResponse(
            items=users,
            total=100,
            page=1,
            per_page=20
        )
    """
    items: List[T]
    total: int
    page: int
    per_page: int
    pages: int
    
    @classmethod
    def create(cls, items: List[T], total: int, page: int, per_page: int):
        """Create paginated response with calculated pages"""
        return cls(
            items=items,
            total=total,
            page=page,
            per_page=per_page,
            pages=(total + per_page - 1) // per_page
        )

def stream_large_dataset(db_query):
    """
    Stream large dataset instead of loading all into memory.
    
    Use for exports, reports, etc.
    
    Example:
        @router.get("/export/users")
        def export_users(db: Session = Depends(get_db)):
            query = db.query(User).yield_per(100)
            return StreamingResponse(
                stream_large_dataset(query),
                media_type="application/x-ndjson"
            )
    """
    def generate():
        for item in db_query:
            yield json.dumps(item.to_dict()) + "\n"
    
    return generate()

import hashlib

def generate_etag(content: str) -> str:
    """Generate ETag from content"""
    return hashlib.md5(content.encode()).hexdigest()


def handle_conditional_request(
    request: Request,
    content: str,
    last_modified: str = None
) -> Response:
    """
    Handle conditional GET requests (304 Not Modified).
    
    Example:
        @router.get("/users/{user_id}")
        def get_user(user_id: int, request: Request):
            user = get_user_from_db(user_id)
            content = user.json()
            
            return handle_conditional_request(
                request,
                content,
                last_modified=user.updated_at.isoformat()
            )
    """
    etag = generate_etag(content)
    
    # Check If-None-Match header
    if request.headers.get("If-None-Match") == etag:
        return Response(status_code=304)  # Not Modified
    
    # Check If-Modified-Since header
    if last_modified and request.headers.get("If-Modified-Since") == last_modified:
        return Response(status_code=304)
    
    # Return full response with ETag
    return Response(
        content=content,
        headers={
            "ETag": etag,
            "Last-Modified": last_modified or "",
            "Cache-Control": "private, must-revalidate"
        }
    )

class PerformanceMonitoringMiddleware(BaseHTTPMiddleware):

    Monitor API performance.

    """
    
    async def dispatch(self, request: Request, call_next: Callable):
        start_time = time.time()
        
        response = await call_next(request)
        
        process_time = time.time() - start_time
        
        # Add timing header
        response.headers["X-Process-Time"] = str(process_time)
        
        # Log slow requests (> 1 second)
        if process_time > 1.0:
            logger.warning(
                f"Slow request: {request.method} {request.url.path} "
                f"took {process_time:.3f}s"
            )
        
        return response
